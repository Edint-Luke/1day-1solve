function solution(a, b, n) {
    let give = a //주면
    let take = b //받고
    let Coke = n //원래 있던 코크

    let result = 0
    let rest = 0

    let answer = 0

    for (i = 0; i <= n; i++) {
        if (Coke) {
            if (Coke < give) {
                Coke = 0
            } else {
                result = Math.trunc(Coke / give) * take
                rest = (Coke - (Math.trunc(Coke / give) * give)) + result
                Coke = rest
                answer = answer + result
            }
        } else {
            return answer
        }

    }
}
